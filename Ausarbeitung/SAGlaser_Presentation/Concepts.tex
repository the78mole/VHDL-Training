\begin{frame}
	\frametitle{Konzeptauswahl}
	\frametitle{Finden geeigneter Algorithmen für das Gesamtsytem}
	\begin{beamerboxesrounded}[shadow=true]{Gesamtsystem bestehend aus:}
		\begin{itemize}[<+->]
			\item Sender bestehend aus:
				\begin{itemize}
					\item (Pseudo-)Zufallszahlengenerator (PRN-Schieberegister)
					\item Signalgenerator (DDS, CORDIC)
					\item Modulator (FSK)
				\end{itemize}
			\item Empfänger bestehend aus:
				\begin{itemize}
					\item Bandpassfilter (1 je Trägerfrequenz)
					\item Hüllkurvendemodulator (1 je Trägerfrequenz)
					\item Tiefpassfilter (1 je Trägerfrequenz)
					\item Vergleicher (mit Hysterese)
				\end{itemize}
			\item Weitere Schnittstellenmodule
			\item Fehlerkorrekturhilfen
		\end{itemize}
	\end{beamerboxesrounded}
\end{frame}
